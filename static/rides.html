// Function to fetch rides from the server and display them async function fetchRides() { const container = document.getElementById("rides-container"); try { const response = await fetch("/rides"); if (!response.ok) { throw new Error("Failed to fetch rides."); } const rides = await response.json(); // Clear existing rides container.innerHTML = ""; // Loop through each ride and create HTML elements rides.forEach(ride => { const rideCard = document.createElement("div"); rideCard.classList.add("ride-box"); rideCard.innerHTML = `

${ride.destination}

Date: ${ride.date}

Time: ${ride.time}

Driver: ${ride.driver_name}

Seats Available: ${ride.seats_available}

Cost per Seat: £${ride.cost}

Book This Ride `; container.appendChild(rideCard); }); } catch (error) { console.error("Error loading rides:", error); container.innerHTML = `

Failed to load rides. Please try again later.

`; } } // Call the function when the page loads document.addEventListener("DOMContentLoaded", fetchRides);
